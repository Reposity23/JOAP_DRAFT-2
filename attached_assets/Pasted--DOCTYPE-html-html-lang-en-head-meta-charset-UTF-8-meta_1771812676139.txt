        chartDonut.data.datasets[0].data = newData.data;
        chartDonut.update();

        // Check if trend is down to change color class
        const visClass = newData.visTrend.includes('↘') ? 'trend-down' : 'trend-up';

        // Update Text for Visitors and Inner Store Circle
        document.getElementById('valVisitors').innerHTML = `${newData.visitors} <span class="${visClass}" style="font-size:11px;">${newData.visTrend}</span>`;
        document.getElementById('valStoreCenter').innerHTML = `${newData.storeCount} <span class="${storeClass}" style="font-size: 10px;">${newData.storeTrend}</span>`;
    });
